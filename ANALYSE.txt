import traci
import csv
# ---------------------------------------------------------
# USER SETTINGS
# ---------------------------------------------------------
SUMO_CFG = "map.sumocfg"
TLS_ID = "6073919354"        # Your traffic light ID
OUTPUT_CSV = "cycle_log.csv"
# ---------------------------------------------------------
# INITIALIZE CSV
# ---------------------------------------------------------
with open(OUTPUT_CSV, "w", newline="") as f:
    writer = csv.writer(f)
    writer.writerow([
        "timestamp",
        "old_phase_index",
        "new_phase_index",
        "old_phase_name",
        "duration_old_phase"
    ])
# ---------------------------------------------------------
# START SUMO
# ---------------------------------------------------------
traci.start(["sumo-gui", "-c", SUMO_CFG])
print(f"Tracking cycle times for traffic light: {TLS_ID}\n")
# ---------------------------------------------------------
# INTERNAL STATE
# ---------------------------------------------------------
prev_phase = traci.trafficlight.getPhase(TLS_ID)
phase_start_time = traci.simulation.getTime()
# OPTIONAL: Give readable names
# Adjust if your TL has more phases
phase_labels = {
    0: "Phase 0 (Green NS)",
    1: "Phase 1 (Yellow NS)",
    2: "Phase 2 (Green EW)",
    3: "Phase 3 (Yellow EW)",
    # Add more if needed
}
# ---------------------------------------------------------
# MAIN LOOP
# ---------------------------------------------------------
try:
    while traci.simulation.getMinExpectedNumber() > 0:
        traci.simulationStep()
        sim_time = traci.simulation.getTime()
        current_phase = traci.trafficlight.getPhase(TLS_ID)
        # -----------------------------
        # Detect PHASE CHANGE
        # -----------------------------
        if current_phase != prev_phase:
            duration = sim_time - phase_start_time
            # Save to CSV
            with open(OUTPUT_CSV, "a", newline="") as f:
                writer = csv.writer(f)
                writer.writerow([
                    sim_time,
                    prev_phase,
                    current_phase,
                    phase_labels.get(prev_phase, f"Phase {prev_phase}"),
                    duration
                ])
            # Print to terminal
            print(
                f"[{sim_time:.2f}s] {phase_labels.get(prev_phase)} "
                f"→ {phase_labels.get(current_phase)} "
                f"(Duration: {duration:.2f}s)"
            )
            # Reset trackers
            prev_phase = current_phase
            phase_start_time = sim_time
finally:
    traci.close()
    print("\nTracking complete. Data saved to:", OUTPUT_CSV)

import React, { useState } from 'react';
import { BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer, LineChart, Line, ComposedChart, Cell } from 'recharts';
import Papa from 'papaparse';

export default function PhaseAnalysisReport() {
  const [data, setData] = useState([]);
  const [report, setReport] = useState(null);

  const handleFileUpload = (e) => {
    const file = e.target.files[0];
    if (!file) return;

    Papa.parse(file, {
      header: true,
      skipEmptyLines: true,
      dynamicTyping: false,
      complete: (results) => {
        const parsed = results.data.map(row => ({
          timestamp: parseFloat(row.timestamp) || 0,
          phase_index: parseInt(row.phase_index) || 0,
          group: row.group || '',
          lanes: row.lanes || '',
          phase_type: row.phase_type || '',
          duration: parseFloat(row.duration) || 0,
          expected_duration: parseInt(row.expected_duration) || 0,
          deviation: parseFloat(row.deviation) || 0
        })).filter(row => row.duration > 0);

        setData(parsed);
        generateReport(parsed);
      },
      error: (err) => {
        alert('Error parsing CSV: ' + err.message);
      }
    });
  };

  const generateReport = (parsed) => {
    if (!parsed || parsed.length === 0) {
      alert('No valid data found in CSV');
      return;
    }

    // Filter only GREEN phases
    const greenPhases = parsed.filter(p => p.phase_type === 'GREEN');
    
    if (greenPhases.length === 0) {
      alert('No GREEN phases found in data');
      return;
    }

    const groupStats = {};
    greenPhases.forEach(phase => {
      if (!groupStats[phase.group]) {
        groupStats[phase.group] = {
          group: phase.group,
          lanes: phase.lanes,
          durations: [],
          deviations: [],
          expected: phase.expected_duration
        };
      }
      groupStats[phase.group].durations.push(phase.duration);
      groupStats[phase.group].deviations.push(phase.deviation);
    });

    // Calculate statistics
    const stats = Object.values(groupStats).map(g => {
      if (g.durations.length === 0) return null;
      
      const mean = g.durations.reduce((a, b) => a + b, 0) / g.durations.length;
      const variance = g.durations.reduce((sum, d) => sum + Math.pow(d - mean, 2), 0) / g.durations.length;
      const std = Math.sqrt(variance);
      const cv = (std / mean) * 100;
      const minDev = Math.min(...g.deviations);
      const maxDev = Math.max(...g.deviations);
      const meanDev = g.deviations.reduce((a, b) => a + b, 0) / g.deviations.length;

      return {
        group: g.group,
        lanes: g.lanes,
        expected: g.expected,
        count: g.durations.length,
        mean: parseFloat(mean.toFixed(2)),
        std: parseFloat(std.toFixed(2)),
        cv: parseFloat(cv.toFixed(1)),
        min: parseFloat(Math.min(...g.durations).toFixed(2)),
        max: parseFloat(Math.max(...g.durations).toFixed(2)),
        minDev: parseFloat(minDev.toFixed(2)),
        maxDev: parseFloat(maxDev.toFixed(2)),
        meanDev: parseFloat(meanDev.toFixed(2))
      };
    }).filter(s => s !== null);

    const totalGreenCycles = greenPhases.length;
    const avgCV = stats.length > 0 
      ? (stats.reduce((sum, s) => sum + s.cv, 0) / stats.length).toFixed(1)
      : '0';
    const isFixedTime = parseFloat(avgCV) < 15;

    setReport({
      stats,
      totalGreenCycles,
      avgCV,
      isFixedTime,
      greenPhases
    });
  };

        {/* HEADER */}

            🚦 Traffic Light Phase Analysis Report
          </h1>
          <p style={{ color: '#666', margin: '0', fontSize: '14px' }}>
            SUMO Simulation - {report.totalGreenCycles} Green Phase Cycles Analyzed
          </p>

        <div style={{
          backgroundColor: report.isFixedTime ? '#d4edda' : '#fff3cd',
